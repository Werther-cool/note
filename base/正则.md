
## 截取url 获取nav
function GetQueryString(name)
{
    var reg = new RegExp("(^|&)"+ name +"=([^&]*)(&|$)");
    var r = window.location.search.substr(1).match(reg);
    if(r!=null)return  unescape(r[2]); return null;
}
if (GetQueryString("nav")) {
    vm.nav =GetQueryString("nav")
}  

(^|&) 以空或者&开头的 字符
name  拼接name
=([^&*]) 拼接= 非&字符n个
(&|$) 直到以 &或者空 结束
